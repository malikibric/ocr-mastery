# Native Tesseract OCR Migration Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Replace the `tesseract.js` WASM dependency with the native Tesseract OCR engine (https://github.com/tesseract-ocr/tesseract), invoked via its CLI, and trim everything that becomes obsolete after the switch.

**Architecture:** OCR helper scripts (`scripts/extract-image-text.cjs`, `scripts/extract-image-layout.cjs`) currently call `tesseract.js`'s `createWorker` and read `data.text` / `data.words`. After this plan they will spawn the system `tesseract` binary via `execFile`, parse plain-text stdout for the text-only path, and parse TSV stdout for the layout path (one row per word, with `left/top/width/height/conf/text` columns). Sharp preprocessing (grayscale, normalize) stays. The `tesseract.js` npm package and the `osd.traineddata` file are removed; only `eng.traineddata` remains and is loaded through `--tessdata-dir`. PDF text extraction is untouched.

**Tech Stack:** Node.js child_process, native `tesseract` CLI (Leptonica + Tesseract 5.x), `sharp` (preprocessing), `eng.traineddata` (LSTM model), Vitest.

## File Structure

- **Modify** `scripts/extract-image-text.cjs` — replace `tesseract.js` worker with `execFile("tesseract", …)` returning plain text on stdout. PSM 6, `preserve_interword_spaces=1`.
- **Modify** `scripts/extract-image-layout.cjs` — replace `tesseract.js` worker with `execFile("tesseract", …, "tsv")`, parse TSV into `{ width, height, words: [{ text, confidence, bbox }] }`. PSM 11, `preserve_interword_spaces=1`.
- **Create** `scripts/lib/run-tesseract.cjs` — shared helper: locate binary, build args, invoke, return stdout. Single source of truth for CLI invocation, error messages, and `--tessdata-dir` resolution.

```js
const { execFile } = require("node:child_process");
const { promisify } = require("node:util");

const execFileAsync = promisify(execFile);

function buildArgs(inputPath, format, options) {
  const { psm, tessdataDir } = options;
  return [
    inputPath,
    "stdout",
    "--tessdata-dir",
    tessdataDir,
    "-l",
    "eng",
    "--psm",
    String(psm),
    "-c",
    "preserve_interword_spaces=1",
    format
  ];
}

async function runTesseract(inputPath, format, options) {
  const args = buildArgs(inputPath, format, options);
  const { stdout } = await execFileAsync("tesseract", args, {
    maxBuffer: 20 * 1024 * 1024,
    timeout: 60_000
  });
  return stdout;
}

module.exports = { buildArgs, runTesseract };
```

```js
const sharp = require("sharp");
const os = require("node:os");
const path = require("node:path");
const fs = require("node:fs");
const { runTesseract } = require("./lib/run-tesseract.cjs");

const TESSDATA_DIR = process.cwd();

async function preprocessImage(filePath) {
  const tmpPath = path.join(os.tmpdir(), `ocr_layout_${process.pid}_${Date.now()}.png`);
  await sharp(filePath).grayscale().normalize().png().toFile(tmpPath);
  return tmpPath;
}

function parseTsv(tsv, width, height) {
  const lines = tsv.split(/\r?\n/);
  const words = [];
  for (let i = 1; i < lines.length; i++) {
    const line = lines[i];
    if (!line) continue;
    const cols = line.split("\t");
    if (cols.length < 12) continue;
    const level = Number(cols[0]);
    if (level !== 5) continue; // word-level rows only
    const text = cols.slice(11).join("\t").trim();
    if (!text) continue;
    const left = Number(cols[6]);
    const top = Number(cols[7]);
    const w = Number(cols[8]);
    const h = Number(cols[9]);
    const conf = Number(cols[10]);
    if (!Number.isFinite(left) || !Number.isFinite(top) || !Number.isFinite(w) || !Number.isFinite(h)) continue;
    words.push({
      text,
      confidence: Number.isFinite(conf) ? conf : 0,
      bbox: { x0: left, y0: top, x1: left + w, y1: top + h }
    });
  }
  return { width, height, words };
}

async function main() {
  const filePath = process.argv[2];
  if (!filePath) throw new Error("An image file path is required.");

  let preprocessed = null;
  try {
    preprocessed = await preprocessImage(filePath);
  } catch {
    preprocessed = filePath;
  }

  try {
    const metadata = await sharp(preprocessed).metadata();
    const tsv = await runTesseract(preprocessed, "tsv", {
      psm: "11",
      tessdataDir: TESSDATA_DIR
    });
    const layout = parseTsv(tsv, metadata.width ?? 0, metadata.height ?? 0);
    process.stdout.write(JSON.stringify(layout));
  } finally {
    if (preprocessed !== filePath && fs.existsSync(preprocessed)) {
      fs.unlinkSync(preprocessed);
    }
  }
}

if (require.main === module) {
  main().catch((error) => {
    process.stderr.write(`${error instanceof Error ? error.message : String(error)}\n`);
    process.exit(1);
  });
}

module.exports = { parseTsv };
```
